git config --list//查看用户配置信息
git init//初始化仓库
git add//将暂存区文件修改提交到版本库中		用法例子：git add +(文件名)
git commit//提交改变		用法例子：git commit -m"........." + (文件名)
git status//查看当前本地仓库状态
git log//查看日志
git reset//将暂存区文件取消修改或者是切到指定版本 --hard(版本号)
git remote//查看远程仓库		-v//查看详细
git remote add//添加远程库		用法例子：git remote add <shortname> <url>
git clone//从远程仓库克隆
git pull//从远程仓库拉取
git push//推送到远程仓库
